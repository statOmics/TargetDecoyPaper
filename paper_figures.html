3.2 FDR function

Calculate FDR on tibble with search results. It is assumed that higher scores are better.

# FDR function
fdrDb <- function(db)
{
  db <- db %>% 
    arrange(desc(score)) %>% 
    mutate(FP = cumsum(is_decoy),
           FDR = cumsum(is_decoy)/cumsum(!is_decoy))
  FDR <- db$FDR
  FDRmin <- FDR[length(FDR)]
  for (j in (length(FDR)-1):1)
  {
    if (FDR[j] < FDRmin)
      FDRmin <- FDR[j] else
      FDR[j] <- FDRmin   
  }
  db$FDR <- FDR
  db <- db %>% 
    arrange(spectrum_id) 
  return(db)
}
